pipeline {
  agent any
  stages {
    stage('Clone repo') {
      steps {
        git 'https://github.com/ton_user/resto_revenue_mlops.git'
      }
    }
    stage('Install dependencies') {
      steps {
        sh 'pip install -r requirements.txt'
      }
    }
    stage('Data preprocessing') {
      steps {
        sh 'python src/preprocessing.py'
      }
    }
    stage('Train model') {
      steps {
        sh 'python src/train_model.py'
      }
    }
    stage('Evaluate model') {
      steps {
        sh 'python src/evaluate_model.py'
      }
    }
    stage('Deploy (optional)') {
      steps {
        sh 'python src/deploy_model.py'
      }
    }
  }
  triggers {
    githubPush()  // Déclenchement automatique à chaque git push
  }
}
